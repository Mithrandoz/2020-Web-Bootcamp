-A year is a leap year if it is evenly divisible by 4

-Unless that year is also evenly divisible by 400

Solution:

function isLeap(year) {
    
    if(year % 4 === 0){
        if(year % 100 === 0){
            if(year % 400 === 0){
                return("Leap year.")
            } else{
                return("Not leap year.")
            }
        } else{
            return("Leap year.")
        }
    } else{
        return("Not leap year.")
    }

}
